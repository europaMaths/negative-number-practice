  <div class="feedback" id="solution"></div>

  <script>
    let currentQuestion = {};
    let attempts = 0;

    function getSelectedOperations() {
      return Array.from(document.querySelectorAll('input[type="checkbox"]:checked'))
        .map(cb => cb.value);
    }

    function getRandomInt(min, max) {
      return Math.floor(Math.random() * (max - min + 1)) + min;
    }

    function generateDivisionOperands(min, max) {
      let divisor = getRandomInt(min, max);
      if (divisor === 0) divisor = 1;
      let result = getRandomInt(min, max);
      let dividend = divisor * result;
      return [dividend, divisor];
    }

    function generateQuestion() {
      const operations = getSelectedOperations();
      const min = parseInt(document.getElementById('minValue').value);
      const max = parseInt(document.getElementById('maxValue').value);
      const op = operations[Math.floor(Math.random() * operations.length)];

      let a, b;
      if (op === '/') {
        [a, b] = generateDivisionOperands(min, max);
      } else {
        a = getRandomInt(min, max);
        b = getRandomInt(min, max);
      }

      currentQuestion = { a, b, op, answer: eval(`${a} ${op} ${b}`) };
      document.getElementById('question').textContent = `What is (${a}) ${op} (${b}) ?`;
      document.getElementById('feedback').textContent = '';
      document.getElementById('solution').textContent = '';
      document.getElementById('giveUpBtn').classList.add('hidden');
      document.getElementById('userAnswer').value = '';
      attempts = 0;
    }

    function checkAnswer() {
      const userAnswer = Number(document.getElementById('userAnswer').value);
      if (isNaN(userAnswer)) {
        document.getElementById('feedback').textContent = 'Please enter a valid number.';
        return;
      }

      attempts++;

      if (userAnswer === currentQuestion.answer) {
        document.getElementById('feedback').textContent = 'âœ… Correct!';
        document.getElementById('giveUpBtn').classList.add('hidden');
      } else {
        document.getElementById('feedback').textContent = 'âŒ Incorrect. Try again.';
        if (attempts >= 3) {
          document.getElementById('giveUpBtn').classList.remove('hidden');
        }
      }
    }

    function showSolution() {
      const { a, b, op, answer } = currentQuestion;
      let explanation = '';
      switch (op) {
        case '+':
          explanation = `Add ${a} and ${b}: ${a} + ${b} = ${answer}`;
          break;
        case '-':
          explanation = `Subtract ${b} from ${a}: ${a} - ${b} = ${answer}`;
          break;
        case '*':
          explanation = `Multiply ${a} and ${b}: ${a} Ã— ${b} = ${answer}`;
          break;
        case '/':
          explanation = `Divide ${a} by ${b}: ${a} Ã· ${b} = ${answer}`;
          break;
      }
      document.getElementById('solution').textContent = `ðŸ’¡ Solution: ${explanation}`;
      document.getElementById('feedback').textContent = '';
      document.getElementById('giveUpBtn').classList.add('hidden');
    }
  </script>
